        // SEARCH & FILTER
        function heroSearch() {
            const dest = document.getElementById('heroDest').value.toLowerCase();
            const dur = document.getElementById('heroDuration').value;
            const price = document.getElementById('heroPrice').value;

        function filterPackages() {
            const dest = document.getElementById('filterDest').value.toLowerCase();
            const dur = document.getElementById('filterDuration').value;
            const price = document.getElementById('filterPrice').value;
            
            const filtered = packages.filter(p => {
                const matchDest = !dest || p.destination.toLowerCase().includes(dest) || p.name.toLowerCase().includes(dest);
                const matchDur = !dur || p.duration == dur;
                const matchPrice = !price || p.price <= parseInt(price);
                return matchDest && matchDur && matchPrice;
            });
            
            const grid = document.getElementById('allPackages');
            if (filtered.length === 0) {
                grid.innerHTML = `
                    <div class="empty-state" style="grid-column:1/-1;">
                        <i class="fas fa-search"></i>
                        <h3>No packages found</h3>
                        <p>Try adjusting your search criteria</p>
                    </div>
                `;
            } else {
                grid.innerHTML = filtered.map(p => createPackageCard(p)).join('');
            }
        }

        // AUTH MODALS
        function openModal(type) {
            document.getElementById('modalOverlay').classList.add('active');
            document.querySelectorAll('.modal').forEach(m => m.style.display = 'none');
            document.getElementById(type + 'Modal').style.display = 'block';
            
            // Reset forms
            document.querySelectorAll('.error-message').forEach(e => e.style.display = 'none');
            document.querySelectorAll('.alert').forEach(a => a.style.display = 'none');
        }

        function closeModal() {
            document.getElementById('modalOverlay').classList.remove('active');
        }

        function switchModal(type) {
            openModal(type);
        }

        function handleLogin(e) {
            e.preventDefault();
            const email = document.getElementById('loginEmail').value;
            const isStaff = document.getElementById('roleSelect').checked;
            
            if (!email.includes('@') || document.getElementById('loginPassword').value.length < 4) {
                document.getElementById('loginError').style.display = 'block';
                return;
            }
            
            // Simulate role-based login
            if (email.includes('admin')) currentRole = 'admin';
            else if (isStaff || email.includes('staff')) currentRole = 'staff';
            else currentRole = 'customer';
            
            currentUser = {name: email.split('@')[0], email: email};
            
            document.getElementById('navAuth').style.display = 'none';
            document.getElementById('navUser').style.display = 'flex';
            document.getElementById('userName').textContent = currentUser.name;
            document.getElementById('navDashboard').style.display = 'block';
            document.getElementById('custName').textContent = currentUser.name;
            
            closeModal();
            showToast('Welcome back, ' + currentUser.name + '!');
            
            if (currentRole !== 'customer') showDashboard();
        }

        function handleRegister(e) {
            e.preventDefault();
            const pass = document.getElementById('regPassword').value;
            const confirm = document.getElementById('regConfirm').value;
            const email = document.getElementById('regEmail').value;
            
            let valid = true;
            
            if (!email.includes('@')) {
                document.getElementById('regEmailError').style.display = 'block';
                valid = false;
            } else {
                document.getElementById('regEmailError').style.display = 'none';
            }
            
            if (pass !== confirm || pass.length < 6) {
                document.getElementById('regPassError').textContent = pass !== confirm ? 'Passwords do not match' : 'Password must be at least 6 characters';
                document.getElementById('regPassError').style.display = 'block';
                valid = false;
            } else {
                document.getElementById('regPassError').style.display = 'none';
            }
            
            if (!valid) {
                document.getElementById('regError').style.display = 'block';
                return;
            }
            
            document.getElementById('regError').style.display = 'none';
            document.getElementById('regSuccess').style.display = 'block';
            
            setTimeout(() => {
                closeModal();
                openModal('login');
            }, 1500);
        }

        function logout() {
            currentUser = null;
            currentRole = 'customer';
            document.getElementById('navAuth').style.display = 'flex';
            document.getElementById('navUser').style.display = 'none';
            document.getElementById('navDashboard').style.display = 'none';
            showPage('home');
            showToast('Logged out successfully');
        }

        // BOOKING
        function openBookingModal() {
            if (!currentUser) {
                openModal('login');
                return;
            }
            document.getElementById('bookPackage').value = selectedPackage ? selectedPackage.name : '';
            document.getElementById('bookName').value = currentUser.name;
            document.getElementById('bookEmail').value = currentUser.email;
            calcTotal();
            openModal('booking');
        }

        function calcTotal() {
            const travelers = parseInt(document.getElementById('bookTravelers').value) || 1;
            const price = selectedPackage ? selectedPackage.price : 0;
            document.getElementById('bookTotal').textContent = '$' + (travelers * price);
        }

        function handleBooking(e) {
            e.preventDefault();
            const ref = 'GT-' + Math.floor(10000 + Math.random() * 90000);
            document.getElementById('bookingRef').textContent = ref;
            document.getElementById('bookingSuccess').style.display = 'block';
            
            setTimeout(() => {
                closeModal();

            }, 2000);
        }

            document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));

        }

        function moveSlide(dir) {
            const slider = document.getElementById('testimonialSlider');
            const cards = slider.children;
            currentSlide = (currentSlide + dir + cards.length) % cards.length;
            slider.style.transform = `translateX(-${currentSlide * 100}%)`;
        }

        // AUTO SLIDE
        setInterval(() => moveSlide(1), 5000);

        function showToast(msg) {
            const toast = document.getElementById('toast');

        }
